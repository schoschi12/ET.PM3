249/** ***************************************************************************
250 * @brief Disable the GYRO on the microcontroller board.
251 *
252 * @note MISO of the GYRO is connected to PF8 and CS to PC1.
253 * @n Some times the GYRO goes into an undefined mode at startup
254 * and pulls the MISO low or high thus blocking the analog input on PF8.
255 * @n The simplest solution is to pull the CS of the GYRO low for a short while
256 * which is done with the code below.
257 * @n PF8 is also reconfigured.
258 * @n An other solution would be to remove the GYRO
259 * from the microcontroller board by unsoldering it.
260 *****************************************************************************/
261static void gyro_disable(void) {
262 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
263 /* Disable PC1 and PF8 first */
264 GPIOC->MODER &= ~GPIO_MODER_MODER1; // Reset mode for PC1
265 GPIOC->MODER |= GPIO_MODER_MODER1_0; // Set PC1 as output
266 GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
267 HAL_Delay(10); // Wait some time
268 GPIOC->MODER |= GPIO_MODER_MODER1_Msk; // Analog mode PC1 = ADC123_IN11
269 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
270 GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8; // Reset speed of PF8
271 GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8; // Reset alternate func. of PF8
272 GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8; // Reset pulup/down of PF8
273 HAL_Delay(10); // Wait some time
274 GPIOF->MODER |= GPIO_MODER_MODER8_Msk; // Analog mode for PF6 = ADC3_IN4
275}

361void tim_TIM7_periodicConfig(uint32_t msPeriod) {
362 //Enable TIM7 timer
363 RCC->APB1ENR |= RCC_APB1ENR_TIM7EN;
364 //1 Pulse mode enable
365 TIM7->CR1 &= ~(TIM_CR1_OPM);
366 //Mode --> RESET
367 TIM7->CR2 &= ~(TIM_CR2_MMS);
368
369 //Prescaler
370 TIM7->PSC = (msPeriod * 10) - 1;
371 //Period
372 TIM7->ARR = 8400 - 1; //->10kHz
373
374 //Clear Update Interrupt
375 TIM7->SR &= ~(1UL << 0);
376 //Enable update interrupt
377 TIM7->DIER |= 0x01;
378 //Enable NVIC Interrupt
379 NVIC_EnableIRQ(TIM7_IRQn);
380 //Start perodic timer
381 TIM7->CR1 |= 0x01;
382}
