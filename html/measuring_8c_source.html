62/** ***************************************************************************
63 * @brief Configure GPIOs in analog mode.
64 *
65 * @note The input number for the ADCs is not equal to the GPIO pin number!
66 * - ADC3_IN4 = GPIO PF6
67 * - ADC123_IN13 = GPIO PC3
68 * - ADC12_IN5 = GPIO PA5
69 * - DAC_OUT2 = GPIO PA5 (= same GPIO as ADC12_IN5)

71void MEAS_GPIO_analog_init(void) {
72 __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
73 GPIOF->MODER |= (GPIO_MODER_MODER6_Msk);// Analog mode for PF6 = ADC3_IN4
74 __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
75 GPIOC->MODER |= (GPIO_MODER_MODER3_Msk);// Analog mode for PC3 = ADC123_IN13
76 __HAL_RCC_GPIOA_CLK_ENABLE(); // Enable Clock for GPIO port A
77 GPIOA->MODER |= (GPIO_MODER_MODER5_Msk); // Analog mode for PA5 ADC12_IN5
78}

107/** ***************************************************************************
108 * @brief Initialize ADC, timer and DMA for sequential acquisition = scan mode
109 *
110 * Uses ADC3 and DMA2_Stream1 channel2
111 * @n The ADC3 trigger is set to TIM2 TRGO event
112 * @n At each trigger both inputs are converted sequentially
113 * and transfered to memory by the DMA.
114 * @n As each conversion triggers the DMA, the number of transfers is doubled.
115 * @n The DMA triggers the transfer complete interrupt when all data is ready.
116 * @n The inputs used are ADC123_IN11 = GPIO PC1 and ADC3_IN6 = GPIO PF8
117 *****************************************************************************/
118void ADC3_scan_init(unsigned long channel1, unsigned long channel2) {
119 MEAS_input_count = 2; // Only 1 input is converted
120 __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
121 ADC3->SQR1 |= ADC_SQR1_L_0; // Convert 2 inputs
122 ADC3->SQR3 |= (channel1 << ADC_SQR3_SQ1_Pos);// Input 13 = first conversion
123 ADC3->SQR3 |= (channel2 << ADC_SQR3_SQ2_Pos);// Input 4 = second conversion
124 ADC3->CR1 |= ADC_CR1_SCAN; // Enable scan mode
125 ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
126 ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
127 ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
128 __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
129 DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
130 while (DMA2_Stream1->CR & DMA_SxCR_EN) {
131 ;
132 } // Wait for DMA to finish
133 DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
134 DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
135 DMA2_Stream1->CR |= DMA_SxCR_PL_1; // Priority high
136 DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
137 DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
138 DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
139 DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
140 DMA2_Stream1->NDTR = 2 * ADC_NUMS; // Number of data items to transfer
141 DMA2_Stream1->PAR = (uint32_t) &ADC3->DR; // Peripheral register address
142 DMA2_Stream1->M0AR = (uint32_t) ADC_samples;// Buffer memory loc. address
143
144}

233/** ***************************************************************************
234 * @brief Interrupt handler for DMA2 Stream4
235 *
236 * Here the interrupt handler is used together with ADC1 and ADC2
237 * in dual mode where they sample simultaneously.
238 * @n The samples from both ADCs packed in a 32 bit word have been transfered
239 * to memory by the DMA2 and are ready for unpacking.
240 * @note In dual ADC mode two values are combined (packed) in a single uint32_t
241 * ADC_CDR[31:0] = ADC2_DR[15:0] | ADC1_DR[15:0]
242 * and are therefore extracted before further processing.
243 *****************************************************************************/
244void DMA2_Stream4_IRQHandler(void) {
245 if (DMA2->HISR & DMA_HISR_TCIF4) { // Stream4 transfer compl. interrupt f.
246 NVIC_DisableIRQ(DMA2_Stream4_IRQn); // Disable DMA interrupt in the NVIC
247 NVIC_ClearPendingIRQ(DMA2_Stream4_IRQn);// Clear pending DMA interrupt
248 DMA2_Stream4->CR &= ~DMA_SxCR_EN; // Disable the DMA
249 while (DMA2_Stream4->CR & DMA_SxCR_EN) {
250 ;
251 } // Wait for DMA to finish
252 DMA2->HIFCR |= DMA_HIFCR_CTCIF4;// Clear transfer complete interrupt fl.
253 TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
254 ADC1->CR2 &= ~ADC_CR2_ADON; // Disable ADC1
255 ADC2->CR2 &= ~ADC_CR2_ADON; // Disable ADC2
256 ADC->CCR &= ~ADC_CCR_DMA_1; // Disable DMA mode
257 /* Extract combined samples */
258 for (int32_t i = ADC_NUMS - 1; i >= 0; i--) {
259 ADC_samples[2 * i + 1] = (ADC_samples[i] >> 16);
260 ADC_samples[2 * i] = (ADC_samples[i] & 0xffff);
261 }
262 ADC_reset();
263 MEAS_data_ready = true;
264 }

267/**
268 * @brief Applies 5 point moving average to values in ADC_DMA buffer.
269 *
270 * Calculates difference between maximum and minimum value.
271 *
272 * @param pointer where amplitude of left sensing pad is to be stored.
273 * @param pointer where amplitude of right sensing pad is to be stored.
274 */
275void MEAS_average(uint32_t *avg_left, uint32_t *avg_right) {
276 uint32_t *ADC_samples = get_ADC_samples();
277 uint32_t left_max = 0;
278 uint32_t right_max = 0;
279 uint32_t left_min = 5000;
280 uint32_t right_min = 5000;
281 double left_value, right_value;
282 int mov_avg = 4;
283 for (int i = 2; i < (get_ADC_NUMS() - 2); i++) {
284 left_value = (ADC_samples[2 * (i - 2) + 1]
285 + ADC_samples[2 * (i - 1) + 1] + ADC_samples[2 * i + 1]
286 + ADC_samples[2 * (i + 1) + 1] + ADC_samples[2 * (i + 2) + 1])
287 / 5;
288 right_value = (ADC_samples[2 * (i - 2)] + ADC_samples[2 * (i - 1)]
289 + ADC_samples[2 * i] + ADC_samples[2 * (i + 1)]
290 + ADC_samples[2 * (i + 2)]) / 5;
291
292 if ((uint32_t) left_value > left_max) {
293 left_max = (uint32_t) left_value;
294 }
295 if ((uint32_t) right_value > right_max) {
296 right_max = (uint32_t) right_value;
297 }
298 if ((uint32_t) left_value < left_min) {
299 left_min = (uint32_t) left_value;
300 }
301 if ((uint32_t) right_value < right_min) {
302 right_min = (uint32_t) right_value;
303 }
304
305 }
306 *avg_left += (left_max - left_min);
307 *avg_right += (right_max - right_min);
308}

310/**
311 * @brief Calculates the standard deviation of the values in the ADC_DMA buffer.
312 *
313 * SD = sqrt(sum(x_i - mean)^2)
314 *
315 * @return double with standard deviation of stored values from ADC_DMA buffer.
316 */
317
318double calculateSD(uint32_t *buffer, uint8_t repetitions) {
319 double sum = 0.0, mean, SD = 0.0;
320 uint8_t i;
321
322 for (i = 0; i < repetitions; ++i) {
323 sum += (double) buffer[i];
324 }
325 mean = sum / repetitions;
326
327 for (i = 0; i < repetitions; ++i) {
328 SD += pow((double) buffer[i] - mean, 2);
329 }
330 return sqrt(SD / repetitions);
331}
