45/**
46 * @brief calculates distance and standarddeviation based on calibration table.
47 *
48 * Averages the amplitude of both sensing pads and searches for corresponding distance. Saves distance in global variable.
49 *
50 * @param amplitude of left sensing pad
51 * @param amplitude of right sensing pad
52 */
53void calc_distance(uint8_t table_index, uint16_t left, uint16_t right,
54 double *distance) {
55
56 int index = 0;
57 float average = ((float) left + (float) right) / 2 * correction;
58 while (average < amplitudeLUT[table_index][index]
59 && index < (sizeof(distanceLUT) / 2 - 2)) { // find corresponding range of amplitude
60 index++;
61 }
62 if ((amplitudeLUT[table_index][index] - average)
63 < (average - amplitudeLUT[table_index][index + 1])) {
64 *distance = distanceLUT[index];
65 } else {
66 *distance = distanceLUT[index + 1];
67 }
68 SD_calc = (double) SD_raw * (distanceLUT[index + 1] - distanceLUT[index])
69 / (amplitudeLUT[table_index][index]
70 - amplitudeLUT[table_index][index + 1]); // calculate std. deviation in mm
71
72}

135 ;
136 MEAS_data_ready = false;
137 MEAS_average(&avg_left, &avg_right);
138 }
139 avg_right /= repetitions[table_index];
140 avg_left /= repetitions[table_index];

144
145 for (int i = 0; i < (STDDEVSIZE - 1); i++) {
146 buffer[i] = buffer[i + 1];

148 buffer[STDDEVSIZE - 1] = (avg_left + avg_right) / 2;
149 SD_raw = calculateSD(&buffer[0], STDDEVSIZE);
150
