41/**
42 * @brief calculates distance based on calibration table.
43 *
44 * Averages the amplitude of both sensing pads and searches for corresponding distance. Saves distance in global variable.
45 *
46 * @param amplitude of left sensing pad
47 * @param amplitude of right sensing pad
48 */
49void calc_distance(uint16_t left, uint16_t right) {
50 int index = 0;
51 float average = ((float) left + (float) right) / 2 * correction;
52 while (average < amplitudeLUT[index]
53 && index < (sizeof(distanceLUT) / 2 - 2)) {
54 index++;
55 }
56 if ((amplitudeLUT[index] - average) < (average - amplitudeLUT[index + 1])) {
57 distance = distanceLUT[index];
58 } else {
59 distance = distanceLUT[index + 1];
60 }
61}

63/**
64 * @brief calculates angle based on calibration table.
65 *
66 * Calculates the ratio between the amplitudes of the left and right sensing pads. Searches for the corresponding angle in the calibration table. Saves angle in global variable.
67 *
68 * @param amplitude of left sensing pad
69 * @param amplitude of right sensing pad
70 */
71void calc_angle(uint16_t left, uint16_t right) {
72 int index = 0;
73 float factor = (float) left / (float) right;
74 while (factor > factorLUT[index]
75 && index < ((sizeof(factorLUT) / sizeof(factorLUT[0])) - 2)) {
76 index++;
77 }
78 if ((factorLUT[index] - factor) < (factor - factorLUT[index + 1])) {
79 angle = angleLUT[index];
80 } else {
81 angle = angleLUT[index + 1];
82 }
83}
