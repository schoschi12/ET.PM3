24 * @brief measures the current and displays it
25 *
26 * Initiates multiple measurements based on REPETITIONS. Calculates the average

28 * Calculates current in the mains cable. Displays the current.

32double measure_current_HAL(void) {
33 double current = 0;
34 double SD = 0;
35 double SD_right = 0;
36 uint32_t buffer_left[REPETITIONS]={0};
37 uint32_t buffer_right[REPETITIONS]={0};
38
39 uint32_t avg_left = 0;
40 uint32_t avg_right = 0;
41
42for(int j = 0; j<REPETITIONS; j++){ //loop to fill the buffer for std. deviation
43 for (int i = 0; i < REPETITIONS; i++) {
44
45 ADC3_scan_init(11, 6);
46 ADC3_scan_start();
47 while (MEAS_data_ready == false);
48 MEAS_data_ready = false;
49 MEAS_average(&avg_left, &avg_right);
50
51 }
52 avg_left = avg_left / REPETITIONS;
53 avg_right = avg_right / REPETITIONS;
54
55 buffer_left[j] = avg_left;
56 buffer_right[j] = avg_right;
57}
58
59/* at a constant distance from the cable, the relationship between current and
60 * measured voltage is linear.
61 *
62 * The function that describes the relationship is:
63 * current = 0.0023*voltage - 0.4228A
64 */
65 if (avg_left > avg_right) {
66 current = (double) (0.0023 * avg_left - 0.4228);
67 SD = (0.0023 * calculateSD(&buffer_left[0], REPETITIONS) -0.2228);
68 } else {
69 current = (double) (0.0023 * avg_right - 0.4228);
70 SD = (0.0023 * calculateSD(&buffer_right[0], REPETITIONS) -0.2228);
71 }
72
73 char text[16];
74 snprintf(text, 15, "current: %.2lfA", (current));
75 BSP_LCD_DisplayStringAt(0, 100, (uint8_t*) text, LEFT_MODE);
76 snprintf(text, 15, "Std.dev: %.2lfA", (SD));
77 BSP_LCD_DisplayStringAt(0, 120, (uint8_t*) text, LEFT_MODE);
78
79
80 return current;
81
82}
